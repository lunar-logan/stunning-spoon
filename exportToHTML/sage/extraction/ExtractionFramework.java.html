47       public ArrayList<SPOTriplet> getTriples() { 
48           ArrayList<SPOTriplet> triples = new ArrayList<>(); 
49    
50           filterByRelation(dependencies.stream(), (td) -> { 
51               handleNominalSubject(td); 
52    
53               ArrayList<IndexedWord> subjectPhrase = new ArrayList<>(); 
54               subjectPhrase.add(handleCoref(subject)); 
55               subjectPhrase.addAll(getSubjectAttributes(subject)); 
56    
57               ArrayList<IndexedWord> predicatePhrase = new ArrayList<>(); 
58               predicatePhrase.addAll(predicates); 
59               predicates.forEach(p -> predicatePhrase.addAll(getPredicateAttributes(p))); 
60    
61               Collections.sort(subjectPhrase); 
62               Collections.sort(predicatePhrase); 
63    
64    
65               objects.forEach(obj -> { 
66                   ArrayList<IndexedWord> objectPhrase = new ArrayList<>(); 
67                   objectPhrase.add(obj); 
68                   objectPhrase.addAll(findAttributes(obj)); 
69                   Collections.sort(objectPhrase); 
70    
71                   triples.add(new SPOTriplet(Arrays.asList(), subjectPhrase, predicatePhrase, objectPhrase)); 
72               }); 
73    
74    
75           }, "nsubj", "nsubjpass"); 
76    
77           return triples; 
78       }

80       private IndexedWord handleCoref(IndexedWord prp) { 
81           IndexedWord sub = null; 
82           if ((prp.tag().toLowerCase().startsWith("prp") /*&& prp.word().toLowerCase().startsWith("it")*/) || 
83                   prp.tag().toLowerCase().startsWith("w")) { 
84               if (lastSubject != null) { 
85                   sub = lastSubject.makeCopy(); 
86               } 
87           } else { 
88               sub = prp; 
89           } 
90           lastSubject = sub; 
91           MemCache.put("lastSubject", lastSubject); 
92           return lastSubject; 
93       }

102          ArrayList<IndexedWord> attrs = new ArrayList<>();

109      private void handleNominalSubject(TypedDependency td) { 
110          predicates.clear(); 
111          objects.clear(); 
112   
113          IndexedWord gov = td.gov();                             // Get the governor 
114          IndexedWord dep = td.dep();                             // Get the dependent 
115          IndexedWord base = null; 
116          boolean iscop = false; 
117   
118          subject = dep;                                          // Subject in all cases  is the dependent 
119          if (gov.tag().startsWith("VB")) {                       // If the governor is a verb then we've found our predicate 
120              predicates.add(gov); 
121              base = gov; 
122          } else { 
123              TypedDependency cop = getRelation("cop", gov); 
124              if (cop != null) { 
125                  base = handleCopRelation(cop); 
126                  iscop = true; 
127              } 
128          } 
129          findObject(base, iscop); 
130      }

138      private IndexedWord handleCopRelation(TypedDependency cop) { 
139          IndexedWord dep = cop.dep(); 
140          if (dep.tag().startsWith("VB")) {                    // Dependent of cop is a verb, probably the predicate 
141              predicates.add(dep); 
142  //            predicates.add(cop.gov()); 
143   
144              TypedDependency acl = getRelation("acl", cop.gov()); 
145              if (acl != null) { 
146                  L.i("Found acl relation for " + cop.gov()); 
147                  predicates.add(acl.dep()); 
148                  return acl.dep(); 
149              } 
150   
151              return cop.gov(); 
152          } 
153          return null; 
154      }

192      private void addConjunct(IndexedWord object) { 
193          TypedDependency cc = getRelation("cc", object); 
194          if (cc != null) { 
195              String ccWord = cc.dep().word(); 
196              if (ccWord.startsWith("or") || ccWord.startsWith("and")) { 
197                  filterByRelation(dependencies.stream(), td -> { 
198                      if (td.gov().equals(object)) { 
199                          objects.add(td.dep()); 
200                      } 
201                  }, "conj"); 
202              } else { 
203                  L.i("Unknown ccWord: " + ccWord + " found, ignoring"); 
204              } 
205          } else { 
206              L.i("No cc relation for " + object); 
207          } 
208      }

210      private void addCase(IndexedWord nmodGov) { 
211          TypedDependency aCase = getRelation("case", nmodGov); 
212          if (aCase == null) { 
213              L.i("No case relation for " + nmodGov + " found"); 
214          } else { 
215              predicates.add(aCase.dep()); 
216          }

219      /** 
220       * Find and returns the auxiliary of a verb. See <b>aux</b> universal dependency 
221       * for more information. 
222       * 
223       * @param verb 
224       * @return 
225       */ 
226      private ArrayList<IndexedWord> getPredicateAttributes(IndexedWord verb) { 
227          ArrayList<IndexedWord> attrs = new ArrayList<>(); 
228          for (TypedDependency dependency : dependencies) { 
229              if (dependency.gov().equals(verb)) { 
230                  if (dependency.reln().getShortName().startsWith("aux")) { 
231                      attrs.add(dependency.dep()); 
232                  } 
233              } 
234          } 
235          attrs.addAll(getCompoundAndAMod(verb)); 
236          return attrs; 
237      }

240      private ArrayList<IndexedWord> getCompoundAndAMod(IndexedWord ofWord) { 
241          ArrayList<IndexedWord> attrs = new ArrayList<>(); 
242   
243          dependencies.stream() 
244                  .filter(td -> td.gov().equals(ofWord)) 
245                  .filter(td -> { 
246                      String relationName = td.reln().getShortName(); 
247                      return relationName.equalsIgnoreCase("compound") || relationName.equalsIgnoreCase("amod"); 
248                  }) 
249                  .forEach(td -> attrs.add(td.dep())); 
250          return attrs; 
251      }

260      private ArrayList<IndexedWord> findAttributes(IndexedWord word) { 
261          ArrayList<IndexedWord> attrs = new ArrayList<>(); 
262   
263          // Add amod and compound modifiers 
264          attrs.addAll(getCompoundAndAMod(word)); 
265   
266          dependencies.stream() 
267                  .filter(td -> td.gov().equals(word)) 
268                  .filter(td -> td.reln().getShortName().equalsIgnoreCase("nummod") || td.reln().getShortName().equalsIgnoreCase("acl")) 
269                  .forEach(td -> { 
270                      attrs.add(td.dep()); 
271                      attrs.addAll(getCompoundAndAMod(td.dep())); 
272                  }); 
273   
274          if (attrs.isEmpty()) { 
275              // Handle nmod relations 
276              dependencies.stream() 
277                      .filter(td -> td.reln().getShortName().equalsIgnoreCase("nmod")) 
278                      .filter(td -> td.gov().equals(word)) 
279                      .forEach(nmod -> { 
280                          attrs.add(nmod.dep()); 
281                          attrs.addAll(getCompoundAndAMod(nmod.dep())); 
282                          dependencies.stream() 
283                                  .filter(ds -> ds.reln().getShortName().equalsIgnoreCase("case") && ds.gov().equals(nmod.dep())) 
284                                  .findFirst().ifPresent(d -> { 
285                              attrs.add(d.dep()); 
286                              if (d.dep().word().equalsIgnoreCase("between")) { 
287                                  dependencies.stream() 
288                                          .filter(ds -> ds.reln().getShortName().equalsIgnoreCase("conj") && ds.gov().equals(d.gov())) 
289                                          .findFirst().ifPresent(e -> attrs.add(e.dep())); 
290                              } 
291                          }); 
292                      }); 
293          } 
294          return attrs; 
295      }

298      private Stream<TypedDependency> getRelations(String relnName) { 
299          return dependencies 
300                  .stream() 
301                  .filter(td -> td.reln().getShortName().equalsIgnoreCase(relnName)); 
302      }

304      private TypedDependency getRelation(String relnName, IndexedWord governor) { 
305          relnName = relnName.toLowerCase(); 
306   
307          for (TypedDependency dependency : dependencies) { 
308              if (dependency.reln().getShortName().toLowerCase().startsWith(relnName) && 
309                      dependency.gov().equals(governor)) { 
310                  return dependency; 
311              } 
312          } 
313          return null; 
314      }

316      private void filterByRelation(Stream<TypedDependency> stream, Consumer<TypedDependency> callback, String... relns) { 
317          stream.filter(td -> { 
318              String shortName = td.reln().getShortName(); 
319              for (String r : relns) { 
320                  if (shortName.startsWith(r)) return true; 
321              } 
322              return false; 
323          }).forEach(callback);                                   // All the filtered relations consumed by the callback 
324      }
